- So the failure happens because to_class does not map a runtime string value 'Test' to its type str, and it does not return the class object type(None) for None.

   def to_class(values):
       result = []
       for v in values:
           # If the caller gave an actual class object, keep it
           if isinstance(v, type):
               result.append(v)
               continue

           # If the caller gave None, return the class type(None)
           if v is None:
               result.append(type(None))
               continue

           # If the caller gave an instance (str, int, float, etc.), return its type
           if not isinstance(v, str):
               result.append(type(v))
               continue

           # If the caller gave a string and you mean "map builtin/basic names" -> map explicitly
           builtin_map = {'str': str, 'int': int, 'float': float, 'NoneType': type(None)}
           if v in builtin_map:
               result.append(builtin_map[v])
               continue

           # If the string is a fully qualified class name, attempt import/resolve
           try:
               module_name, _, class_name = v.rpartition('.')
               if module_name:
                   mod = __import__(module_name, fromlist=[class_name])
                   result.append(getattr(mod, class_name))
               else:
                   # fallback: maybe they named a builtin/class in global scope
                   result.append(globals().get(v, v))  # or raise an informative error
           except Exception as e:
               raise ValueError(f"cannot resolve class name '{v}': {e}")
       return result
