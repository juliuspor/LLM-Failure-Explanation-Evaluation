def to_locale(s):
    if s is None or len(s) == 0:
        raise ValueError("Invalid locale format: {}".format(s))
    parts = s.split('_')  # preserves empty tokens in most languages
    language = parts[0]
    if not language.isalpha() or not (2 <= len(language) <= 3):
        raise ValueError("Invalid language: {}".format(language))
    country = ''
    variant = ''
    if len(parts) == 2:
        country = parts[1]
    elif len(parts) >= 3:
        country = parts[1]  # may be empty
        variant = '_'.join(parts[2:])  # allow underscores inside variant
    return Locale(language, country, variant)
