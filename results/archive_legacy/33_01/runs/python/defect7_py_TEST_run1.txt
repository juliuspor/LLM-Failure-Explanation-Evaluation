2. Implement explicit conversion rules. Example implementation to replace the buggy code:

def to_class(items):
    import builtins, importlib
    result = []
    for item in items:
        # already a class
        if isinstance(item, type):
            result.append(item)
            continue

        # None -> type(None)
        if item is None:
            result.append(type(None))
            continue

        # instances -> their type
        if not isinstance(item, str):
            result.append(type(item))
            continue

        # item is a string: try builtins first, then import by dotted path
        name = item
        if hasattr(builtins, name):
            result.append(getattr(builtins, name))
            continue

        # try dotted module path like 'package.module.ClassName'
        try:
            module_name, class_name = name.rsplit('.', 1)
            module = importlib.import_module(module_name)
            cls = getattr(module, class_name)
            if isinstance(cls, type):
                result.append(cls)
                continue
        except Exception:
            pass

        # if we reach here, resolution failed — raise a clear error
        raise NameError(f"to_class: cannot resolve '{name}' to a class")

    return result
