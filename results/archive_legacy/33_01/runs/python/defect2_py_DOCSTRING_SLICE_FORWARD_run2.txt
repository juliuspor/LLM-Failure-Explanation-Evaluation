Corrected version (explicit, robust mapping and checks)
def paint_for_value(value):
    # Expect numeric input in [0.0, 1.0]; map to 0..255 grayscale
    if not isinstance(value, (int, float)):
        raise TypeError(f"value must be numeric, got {type(value)!r}")
    if value != value:  # NaN check
        raise ValueError("value is NaN")
    # clamp or validate depending on desired behavior; here we validate:
    if value < 0.0 or value > 1.0:
        raise ValueError(f"value {value!r} outside permitted range [0.0, 1.0]")
    # round to nearest int and ensure in 0..255
    channel = int(round(value * 255))
    channel = max(0, min(255, channel))
    return (channel, channel, channel)
