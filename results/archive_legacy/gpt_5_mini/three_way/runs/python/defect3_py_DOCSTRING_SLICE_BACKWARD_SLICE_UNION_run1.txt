def to_utf16_code_units(s: str) -> List[int]:
    code_units = []
    for cp in (ord(ch) for ch in s):
        if cp <= 0xFFFF:
            code_units.append(cp)
        else:
            cp -= 0x10000
            code_units.append(0xD800 + (cp >> 10))
            code_units.append(0xDC00 + (cp & 0x3FF))
    return code_units
