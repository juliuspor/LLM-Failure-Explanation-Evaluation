def get_paint(self, value: float) -> Tuple[int, int, int]:
    # Guard: invalid bounds
    if self._upper_bound == self._lower_bound:
        # choose behavior: treat as black, or raise configuration error
        return (0, 0, 0)

    # Clamp input value to bounds
    if value < self._lower_bound:
        value = self._lower_bound
    elif value > self._upper_bound:
        value = self._upper_bound

    # Compute gray component and clamp to [0,255] just in case
    g = int((value - self._lower_bound) / (self._upper_bound - self._lower_bound) * 255.0)
    g = max(0, min(255, g))
    return (g, g, g)
