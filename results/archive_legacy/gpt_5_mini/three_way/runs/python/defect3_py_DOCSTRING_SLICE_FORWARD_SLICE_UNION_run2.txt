           else:
               # produce surrogate pair
               cp -= 0x10000
               high = 0xD800 + (cp >> 10)
               low = 0xDC00 + (cp & 0x3FF)
               code_units.append(high)
               code_units.append(low)
       return code_units

       # convert each 2 bytes into an int (0..0xFFFF)
