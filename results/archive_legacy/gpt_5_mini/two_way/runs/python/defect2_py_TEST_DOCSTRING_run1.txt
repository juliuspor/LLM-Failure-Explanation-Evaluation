def get_paint(self, value):
    # clamp input value to configured bounds
    if value < self.lower:
        value = self.lower
    elif value > self.upper:
        value = self.upper

    # avoid division by zero if bounds are equal
    if self.upper == self.lower:
        frac = 0.0
    else:
        frac = (value - self.lower) / (self.upper - self.lower)

    component = int(frac * 255.0)
    # safety clamp to 0..255
    component = max(0, min(255, component))
    return (component, component, component)
